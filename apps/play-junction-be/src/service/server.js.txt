// server.js

const express = require('express');
const bodyParser = require('body-parser');

const app = express();
const port = 3000;

app.use(bodyParser.json());

// Endpoint for payment validation
app.post('/validate-payment', (req, res) => {
    // Perform payment validation logic here
    // For simplicity, let's assume payment is valid if the amount is greater than zero
    const { amount } = req.body;
    const isValid = amount > 0;

    // Send response indicating payment validation result
    res.json({ isValid });
});

app.listen(port, () => {
    console.log(`Server is running on http://localhost:${port}`);
});
